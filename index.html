    const results = document.getElementById('results');


    groupSelect.addEventListener('change', () => {

      subjectSelect.innerHTML = '<option value="">Choose Subject</option>';

      levelSelect.innerHTML = '<option value="">Choose Level</option>';

      inputFields.innerHTML = '';

      results.classList.add('hidden');


      if (data[groupSelect.value]) {

        subjectSelect.disabled = false;

        Object.keys(data[groupSelect.value]).forEach(subject => {

          const option = document.createElement('option');

          option.value = subject;

    subjectSelect.addEventListener('change', () => {

      levelSelect.innerHTML = '<option value="">Choose Level</option>';

      inputFields.innerHTML = '';

      results.classList.add('hidden');


      const group = groupSelect.value;

      const subject = subjectSelect.value;


      if (data[group] && data[group][subject]) {

        levelSelect.disabled = false;

        data[group][subject].forEach(level => {

          const option = document.createElement('option');

          option.value = level;

          option.textContent = level;

          levelSelect.appendChild(option);

        });

      } else {

        levelSelect.disabled = true;

      }

    });


    levelSelect.addEventListener('change', () => {

      const courseKey = `${subjectSelect.value} ${levelSelect.value}`;

      const course = courses[courseKey];


      if (course) {

        inputFields.innerHTML = '';

        course.inputs.forEach((input, idx) => {

          inputFields.innerHTML += `

              <label class="block text-sm font-medium text-gray-700">${input.name} (out of ${input.max})</label>

              <input type="number" min="0" max="${input.max}" id="input${idx}" class="w-full p-2 border rounded-lg">

            </div>

          `;

        });


        inputFields.innerHTML += `

        `;


        document.getElementById('calculate').addEventListener('click', () => {

          calculateGrade(course);

        });

      } else {

        inputFields.innerHTML = '<div class="mt-4 text-center text-gray-500">Coming soon...</div>';

      }

    });


    function calculateGrade(course) {

      let total = 0;

      let valid = true;


      course.inputs.forEach((input, idx) => {

        const val = parseFloat(document.getElementById(`input${idx}`).value);

        if (isNaN(val) || val < 0 || val > input.max) valid = false;

        total += (val / input.max) * 100 * input.weight;

      });


      if (!valid) {

        alert('Please enter valid scores.');

        return;

      }


      total = Math.round(total);


      const grade = course.boundaries.find(b => total >= b.min && total <= b.max)?.grade || 0;


      results.classList.remove('hidden');

      results.innerHTML = `<h2 class="text-2xl font-bold text-blue-800">Final Weighted Score: ${total}%</h2><p class="text-lg text-gray-700">Predicted IB Grade: ${grade}</p>`;

    }
